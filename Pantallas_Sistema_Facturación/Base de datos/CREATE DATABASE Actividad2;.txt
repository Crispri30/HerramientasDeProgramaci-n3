CREATE DATABASE Actividad2;
USE Actividad2;

CREATE TABLE Clientes(
	DocumentoCliente INT PRIMARY KEY,
	NombreCliente NVARCHAR (200) NOT NULL,
	Email NVARCHAR (200) NOT NULL,
	Direccion NVARCHAR (200) NOT NULL,
	Telefono INT NOT NULL
);

CREATE TABLE Usuarios(
	Documento INT NOT NULL,
	Usuario NVARCHAR(200) NOT NULL,
	Clave NVARCHAR(200) NOT NULL
);

CREATE TABLE Categorias(
	CodigoCat INT PRIMARY KEY NOT NULL,
	Nombre NVARCHAR(100)
);


CREATE TABLE Productos(
	CodigoCat INT PRIMARY KEY NOT NULL,
	NombreProducto NVARCHAR(100) NOT NULL,
	PrecioCompra INT NOT NULL,
	PrecioVenta INT NOT NULL,
	Categoria INT,
	CantStock INT NOT NULL
	
	CONSTRAINT FK_Categoria FOREIGN KEY (Categoria) REFERENCES Categorias(CodigoCat)
);
